import Head from 'next/head'
import Intro from '../static/Intro.mdx'
import LatestPosts from '../components/LatestPosts.js'
import getPosts from '../lib/getPosts.js'

export const getStaticProps = async () => {
  const posts = await getPosts({ isExternal: false })
  return {
    props: {
      posts: posts.reverse().slice(0, 5)
    }
  }
}

<Head>
  <title>Ian Walter</title>
</Head>

<Intro />

This is my personal website that I built with [Next.js](https://nextjs.org/).
The source code is available on
[GitHub](https://github.com/ianwalter/ianwalter).

<LatestPosts posts={props.posts} />
